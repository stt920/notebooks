	}

	int flag = 0;

		int sum = 0;
		for (int j = 0; j < 4; j++)

			max = sum;

		}
	}
	cout << flag;

{
	string str;
	getline(cin, str);
	bool flag=true;
	for (size_t i = 0; i < str.size(); i++)
	{
		if ((str[i] >= 'a'&&str[i] <= 'z') || (str[i] >= 'A'&&str[i] <= 'Z')) {
			if (str[i] >= 'a'&&str[i] <= 'z')
			{
				str[i] = toupper(str[i]);
				flag = false;
			}
			str[i] = str1[f(str[i])];
			if (flag == false)
				str[i] = tolower(str[i]);
			flag = true;
		}
	}
	cout << str<<endl;
	return 0;
}
